    public static int[ ] sum_array_2(int[ ] a , int[ ] b ){  
  
        int L = Math.max( a.length , b.length );  
        int[ ] res = new int[L];  
  
        int i = a.length;  
        int j = b.length;  
  
  
        i--;  
        j--;  
        L--;  
  
        int  bro = 0 ;  
        while( L>=0 ){  
  
            int diff = (- bro );  
  
            if(i>= 0 ) diff += a[i--];  
  
            if(j>= 0) diff -= b[j--];  
  
            if( diff < 0 ) {  
                diff += 10 ;  
                bro = 1 ;  
            }  
            else {  
                bro = 0 ;  
            }  
  
            res[L] = diff ;  
             
            L--;  
  
        }  
  
        return res;  
  
    }
