import java.util.*;  
 
public class Main {  
    public static void main(String[] args) {  
        Scanner sc = new Scanner(System.in);  
        int n = sc.nextInt( );  
  
        int stars = 1 ;  
        int space = n-1 ;  
  
        for(int i = 1 ; i <= 2*n - 1 ; i++ ){  
  
            for(int j = 1 ; j <= space  ; j++ ){            /// space logic  
                System.out.print("\t");  
            }  
  
            for(int j = 1 ; j<= stars ; j++ ){          /// stars logic  
                System.out.print("*\t");  
            }  
  
            System.out.println( ) ;  
  
            if(i < n ){  
                stars+=2 ;  
                space--;  
            }  
            else{  
                stars-=2;  
                space++;  
            }  
  
        }  
  
    }  
}
